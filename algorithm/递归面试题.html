# 递归面试题

# 1.对象拍平

输入：

{
  a: {
    b: 1,
    c: 2,
    d: {e: 5},

  },
  b: [1, 3, {a: 2, b: 3}],
  c: 3
}

输出：

{
  'a.b': 1,
  'a.c': 2,
  'a.d.e': 5,
  '[b].0': 1,
  '[b].1': 3,
  '[b].2.a': 2,
  '[b].2.b': 3,
  c: 3
}

代码实现

let outObj = {}
const fun = function (cur, k) {
  for(let j in cur) {
    const type = Object.prototype.toString.call(cur[j])
    if(type === '[object Object]') {
      // 是对象
      //也可以使用判断， cur[j] instanceof Object 
      let strKey = k ? `${k}.${j}` : `${j}`
      fun(cur[j], strKey)
    }else if(type === '[object Array]') {
      // 是数组
      let strKey = k ? `${k}[${j}]`: `[${j}]`
      fun(cur[j], strKey)
    }else{
      // 得到最内层值，并进行存储
      let strKey = k ? `${k}.${j}` : `${j}`
      outObj[strKey] = cur[j]
    }
  }
}

let obj = {
  a: {
    b: 1,
    c: 2,
    d: {e: 5},

  },
  b: [1, 3, {a: 2, b: 3}],
  c: 3
}
fun(obj)

console.log('对象拍平', outObj)
